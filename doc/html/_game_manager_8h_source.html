176  case 'R': player->respawn(); break;

178  case 'v': target->setRandomPosition(); break;

180  case 'i':
181  increaseSpeedMultipliers(); break;
182  // Night lights
183  case 'n': directionalLight->toggleLight(); break;
184  case 'h': spotLight->toggleLight(); break;
185  case 'c':
186  for (auto &c: pointLights)
187  c->toggleLight();
188  break;
189  // Stop/Continue game
190  case 's':
191  if (player->currentLives != 0) {
192  isPlaying = !isPlaying;
193  changeAnimationState(isPlaying);
194  }
195  break;
196  case 'r':
197  if (player->currentLives <= 0) {
198  resetSpeedMultipliers();
199  player->restartGame();
200  isPlaying = true;
201  infoString = "";
202  }
203  break;
204 
205  // CameraType
206  case '1': selectCamera(CAMERA_PERSPECTIVE_FOLLOW); break;
207  case '2': selectCamera(CAMERA_PERSPECTIVE_FIXED); break;
208  case '3': selectCamera(CAMERA_ORTHO); break;

212  void movePlayer(float x, float z) {
213  // Check the player movement timeout
214  GLint curTime = glutGet(GLUT_ELAPSED_TIME);
215  if (curTime - lastMoveTime < moveTimeout) {
216  return;
217  } else if(!isPlaying) {
218  return;
219  }
220 
221  Vector3 moveVec = Vector3(x, 0, z);
222  Vector3 tempPos = player->position + moveVec;
223  if (tempPos.z < topVerticalLimitPlayerPos || tempPos.z > bottomVerticalLimitPlayerPos ||
224  tempPos.x > rightHorizontalLimitPlayerPos || tempPos.x < leftHorizontalLimitPlayerPos) {
225  return;
226  }
227 
228  lastMoveTime = curTime;
229  player->jump(moveVec, 0.5f);
